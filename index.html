function calc() { const amtA = parseFloat(document.getElementById('amt-a').value); const prcA = parseFloat(document.getElementById('prc-a').value); const amtB = parseFloat(document.getElementById('amt-b').value); const prcB = parseFloat(document.getElementById('prc-b').value); const cardA = document.getElementById('card-a'); const cardB = document.getElementById('card-b'); const resA = document.getElementById('unit-a'); const resB = document.getElementById('unit-b'); // 1単位あたりの価格を計算 let upA = (amtA > 0 && prcA > 0) ? (prcA / amtA) : null; let upB = (amtB > 0 && prcB > 0) ? (prcB / amtB) : null; // 表示（小数点第4位まで） resA.innerText = upA !== null ? upA.toFixed(4) : "-"; resB.innerText = upB !== null ? upB.toFixed(4) : "-"; // 判定リセット cardA.classList.remove('winner'); cardB.classList.remove('winner'); // 比較ロジック：数値が「小さい」方が WINNER if (upA !== null && upB !== null) { if (upA < upB) { cardA.classList.add('winner'); } else if (upB < upA) { cardB.classList.add('winner'); } // 同値の場合は両方 winner にしない、あるいは片方のみ等、運用に合わせる（現状は判定なし） } }
